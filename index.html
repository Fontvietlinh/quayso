const startTime = audioCtx.currentTime; const osc = audioCtx.createOscillator(); const gain = audioCtx.createGain(); const filter = audioCtx.createBiquadFilter(); // Dùng sóng tam giác để có độ đanh của gỗ/nhựa osc.type = 'triangle'; osc.frequency.setValueAtTime(150, startTime); osc.frequency.exponentialRampToValueAtTime(40, startTime + 0.04); // Filter để tiếng kêu gọn, không bị vang quá filter.type = 'bandpass'; filter.frequency.value = 800; filter.Q.value = 1; gain.gain.setValueAtTime(0.15, startTime); gain.gain.exponentialRampToValueAtTime(0.01, startTime + 0.04); osc.connect(filter); filter.connect(gain); gain.connect(audioCtx.destination); osc.start(); osc.stop(startTime + 0.04); // GIẬT MŨI TÊN (Cải tiến độ nhạy) pointer.style.transition = 'none'; pointer.style.transform = 'translateY(-50%) rotate(20deg)'; // Ép trình duyệt vẽ lại ngay lập tức pointer.offsetHeight; pointer.style.transition = 'transform 0.05s ease-out'; pointer.style.transform = 'translateY(-50%) rotate(0deg)'; } // --- LOGIC THEO DÕI VÒNG QUAY ĐỂ PHÁT ÂM THANH --- function spin() { if (isSpinning || names.length === 0) return; isSpinning = true; const rounds = 8 + Math.floor(Math.random() * 5); const stopAngle = Math.random() * Math.PI * 2; const targetRotation = currentRotation + (rounds * Math.PI * 2) + stopAngle; canvas.style.transition = "transform 7s cubic-bezier(0.15, 0, 0, 1)"; canvas.style.transform = `rotate(${targetRotation}rad)`; let lastSliceIdx = -1; function monitor() { if (!isSpinning) return; const currentRealAngle = getCurrentRotation(); const sliceAngle = (Math.PI * 2) / names.length; const normalized = (Math.PI * 2 - (currentRealAngle % (Math.PI * 2))) % (Math.PI * 2); const currentIdx = Math.floor(normalized / sliceAngle); if (currentIdx !== lastSliceIdx) { playMechanicalTick(); // Gọi âm thanh cơ khí mới lastSliceIdx = currentIdx; } requestAnimationFrame(monitor); } requestAnimationFrame(monitor); setTimeout(() => { isSpinning = false; currentRotation = targetRotation; // ...
